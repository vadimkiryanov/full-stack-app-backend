2. npm i express - установить библиотку express
3. npm i nodemon - для автоматического обновления информации без перезапуска сервера
4. {
    Добавить эти все команды:
     "start": "node index.js",
     "start:dev": "nodemon index.js"
}
5. npm i jsonwebtoken
6. npm i mongoose - позволяет работать с БД MongoDB
